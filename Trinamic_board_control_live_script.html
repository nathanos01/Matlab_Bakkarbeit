Set parameters

% for reference see pictures below
mode = "direct"; % mode selection
v_max = 51200; % maximum velocity [pps]
max_accel = 10000; % acceleration [pps^2]
max_decel = 20000; % deceleration [pps^2]
v1 = 0; % [pps] if set to 0, a1 and d1 are deactivated, only max_accel and max_decel used.
wait_time =1; % [seconds] waiting time after ramping down to v = 0 before next movement
 
% following in position mode only
v_stop = 50; % [pps]
v_start = 50; % [pps]
a1 = 15000; % [pps^2] ignored if v1 = 0
d1 = 30000; % [pps^2] ignored if v1 = 0
wait_time = wait_time / 0.000032;

Data transfer

% open com port for data transfer
fid = serial('COM4','BaudRate',9600, 'DataBits', 8, 'Parity', 'none','StopBits', 1, 'FlowControl', 'none');
fopen (fid);
%send command
fwrite(fid,outcom);
pause(10);
%get answer
answer = fread(fid, 9, 'uint8')
% close com port connection
fclose(fid);

Functions to move motor to positon

% ROR Command
function move_right(nin)
n = int32(nin);
% generate command
% byte1 - adress
outcom(1:4) = uint8([1,4,1,0]);
outcom(5)= uint8(bitand(bitshift(n,-24),255) );
outcom(6)= uint8(bitand(bitshift(n,-16),255) );
outcom(7)= uint8(bitand(bitshift(n,-8),255) );
outcom(8)= uint8(bitand(n,255) );
outcom(9)=uint8(bitand(sum(outcom(1:8)),255) ),
 
% when you enter the function " move_right(1, 0.005,0.01)" in your
% window command, the motor can move.
